**Types:**
```typescript
export type SufficiencyStatus = 'evidence_needed' | 'in_review' | 'sufficient';

export interface CoverageObjective {
  id: string;
  objective_text: string;
  information_need: string | null;
  sort_order: number;
  evidence_count: number;
  sufficiency_status: SufficiencyStatus;
}
```

**`linkEvidenceToObjectives(engagementId, evidenceId, objectiveIds)`:**
- Validate evidenceId exists and belongs to engagement (query evidence_items) — throw 404 if not
- Validate each objectiveId exists and belongs to this engagement's planning record:
  - JOIN: `planning_record_objectives JOIN planning_records ON planning_records.id = planning_record_id WHERE planning_records.engagement_id = :engagementId AND planning_record_objectives.id IN (:objectiveIds)`
  - Throw `{ status: 422, message: 'One or more objective IDs are invalid for this engagement.' }` if any invalid
- For each (evidenceId, objectiveId) pair:
  - Try INSERT INTO `evidence_objective_links` (`id`, `evidence_item_id`, `objective_id`, `created_at`)
  - Use ON CONFLICT (evidence_item_id, objective_id) DO NOTHING
- Count `linked` (rows actually inserted) vs `skipped` (duplicates)
- Return `{ linked, skipped }`

**`unlinkEvidenceFromObjective(engagementId, evidenceId, objectiveId)`:**
- Verify link exists (query evidence_objective_links with both IDs)
- If not found: throw `{ status: 404, message: 'Link not found.' }`
- DELETE FROM `evidence_objective_links` WHERE `evidence_item_id = :evidenceId AND objective_id = :objectiveId`
- Return `{ message: 'Link removed.' }`

**`getObjectiveCoverage(engagementId)`:**
- Get all objectives for this engagement (via planning_records):
  ```sql
  SELECT pro.*, 
    COUNT(eol.id) AS evidence_count,
    COALESCE(os.sufficiency_status, 'evidence_needed') AS sufficiency_status
  FROM planning_record_objectives pro
  JOIN planning_records pr ON pr.id = pro.planning_record_id
  LEFT JOIN evidence_objective_links eol ON eol.objective_id = pro.id
  LEFT JOIN objective_sufficiency os ON os.objective_id = pro.id
  WHERE pr.engagement_id = :engagementId
  GROUP BY pro.id, os.sufficiency_status
  ORDER BY pro.sort_order ASC
  ```
  - NOTE: Check Phase 1 migration 002 for exact table names. The sufficiency status may be stored in the `planning_record_objectives` table itself (as a column like `sufficiency_status`) or in a separate `objective_sufficiency` table. Read `backend/src/db/migrations/` to confirm. Use whichever table exists. If column is on `planning_record_objectives` itself, just SELECT from that table directly.
- Compute: `uncovered_count` = count of objectives where `evidence_count = 0`
- Return `{ objectives: CoverageObjective[], uncovered_count, total: objectives.length, covered: objectives.length - uncovered_count }`

**`setSufficiency(engagementId, updates, actorId)`:**
- `updates`: `Array<{ objective_id: string, sufficiency: SufficiencyStatus }>`
- Validate: each sufficiency value must be 'evidence_needed', 'in_review', or 'sufficient'
- For each update:
  - Check evidence_count for that objective: `SELECT COUNT(*) FROM evidence_objective_links WHERE objective_id = :id`
  - If `sufficiency` is 'in_review' or 'sufficient' AND evidence_count = 0:
    throw `{ status: 422, message: 'Cannot mark objective In Review or Sufficient — it has no linked evidence items.' }`
  - Upsert sufficiency: check the Phase 1 migration for where sufficiency is stored.
    - If column on `planning_record_objectives`: `UPDATE planning_record_objectives SET sufficiency_status = :status, updated_at = NOW() WHERE id = :objective_id`
    - If separate table: INSERT ... ON CONFLICT DO UPDATE
- Write AuditEvent: `INSERT INTO audit_events { engagement_id, action_code: 'OBJECTIVE_SUFFICIENCY_UPDATED', actor_id: actorId, metadata: JSON.stringify({ updates }) }`
- Return `{ updated: updates.length }`

Use `db` from `'../db'`. Follow error shape `{ status, message }` for all throws.

**Role access:**
- POST link: AN, EM, AD (analysts and managers can link)
- DELETE unlink: AN, EM, AD
- GET coverage: all authenticated roles
- PUT sufficiency: QA, EM, AD only (per US-10.2: QA only sets sufficiency; EM/AD also allowed)

**Router structure:**
```typescript
import { Router, Request, Response } from 'express';
import { authenticateSession } from '../middleware/auth';
import { requireRole } from '../middleware/rbac';
import {
  linkEvidenceToObjectives,
  unlinkEvidenceFromObjective,
  getObjectiveCoverage,
  setSufficiency,
} from '../services/objectivecoverage.service';

export const objectiveCoverageRouter = Router({ mergeParams: true });
objectiveCoverageRouter.use(authenticateSession);

// POST /evidence/:evidence_id/objectives — link evidence to objectives
// Mounted under /:id so full path is /api/engagements/:id/evidence/:evidence_id/objectives
objectiveCoverageRouter.post('/evidence/:evidence_id/objectives', requireRole('AN', 'EM', 'AD'), async (req, res) => {
  try {
    const { objective_ids } = req.body;
    if (!Array.isArray(objective_ids) || objective_ids.length === 0) {
      return res.status(422).json({ error: 'objective_ids must be a non-empty array.' });
    }
    const result = await linkEvidenceToObjectives(req.params.id, req.params.evidence_id, objective_ids);
    res.json(result);
  } catch (err: any) {
    if (err.status && err.status < 500) return res.status(err.status).json({ error: err.message });
    console.error('Link objectives error:', err);
    res.status(500).json({ error: 'Internal server error' });
  }
});

// DELETE /evidence/:evidence_id/objectives/:objective_id — unlink
objectiveCoverageRouter.delete('/evidence/:evidence_id/objectives/:objective_id', requireRole('AN', 'EM', 'AD'), async (req, res) => { /* ... */ });

// GET /objectives/coverage — coverage summary
objectiveCoverageRouter.get('/objectives/coverage', async (req, res) => { /* ... */ });

// PUT /objectives/sufficiency — set sufficiency (QA/EM/AD only)
objectiveCoverageRouter.put('/objectives/sufficiency', requireRole('QA', 'EM', 'AD'), async (req, res) => {
  try {
    const updates = req.body; // array of { objective_id, sufficiency }
    if (!Array.isArray(updates) || updates.length === 0) {
      return res.status(422).json({ error: 'Body must be a non-empty array of { objective_id, sufficiency } objects.' });
    }
    const result = await setSufficiency(req.params.id, updates, req.user!.id);
    res.json(result);
  } catch (err: any) {
    if (err.status && err.status < 500) return res.status(err.status).json({ error: err.message });
    console.error('Set sufficiency error:', err);
    res.status(500).json({ error: 'Internal server error' });
  }
});
```
